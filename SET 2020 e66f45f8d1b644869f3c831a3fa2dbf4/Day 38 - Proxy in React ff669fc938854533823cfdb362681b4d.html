Proxy (target,handler)

  • Can thiệp vào object và thay đổi đc nhiều thứ bên trong
  • đề: 1 obj bất kì ⇒ obj.someField ⇒ always trả về null

get: function (target,prop,receiver)

receiver = nguyên cục proxy

const target = {
  name: "minh",
  message: "everyone"
};

const handler4 = {
  get: function(target, prop) {
    if (!(prop in target))
      return null
    return target[prop]
  }
}
const proxy4 = new Proxy(target, handler4)
console.log(proxy4.lon)

let validator = {
  set: function(obj, prop, value) {
    if (prop === 'age') {
      if (!Number.isInteger(value)) {
        throw new TypeError('The age is not an integer');
      }
      if (value > 20 || value < 8) {
        throw new RangeError('The age seems invalid');
      }
    }
    if (prop === 'username') {
      if (value.length > 15 || value.length < 8) {
        throw new RangeError('The username seems invalid');
      }
    }
    if (prop === 'password') {
      if (value.length > 20 || value.length < 8) {
        throw new RangeError('The password seems invalid');
      }
    }
    if (prop === 'email') {
      const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/
      if (!(re.test(value))) {
        throw new RangeError('The email seems invalid');
      }
    }
    // The default behavior to store the value
    return obj[prop] = value;
  }
}
const person = new Proxy({}, validator);
person.age = 19;
console.log(person.age);
person.username = 'tytu1201'
console.log(person.username);
person.email = 'fightforml@gmail.com'
console.log(person.email);
